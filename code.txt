To implement the Fractional Knapsack Problem using the Greedy technique in C.

Code:

#include<stdio.h>
int main(){
int n,i,j;float w[10],p[10],r[10],c,tp=0,t;
printf("Enter number of items: ");
scanf("%d",&n);
printf("Enter weights and profits:\n");
for(i=0;i<n;i++){scanf("%f%f",&w[i],&p[i]);r[i]=p[i]/w[i];}
printf("Enter knapsack capacity: ");
scanf("%f",&c);
for(i=0;i<n;i++)
for(j=i+1;j<n;j++)
if(r[i]<r[j]){
t=r[i];r[i]=r[j];r[j]=t;
t=w[i];w[i]=w[j];w[j]=t;
t=p[i];p[i]=p[j];p[j]=t;}
for(i=0;i<n;i++)
if(c>0&&w[i]<=c){c-=w[i];tp+=p[i];}
else{tp+=r[i]*c;break;}
printf("Maximum Profit: %.2f\n",tp);
}

#include<stdio.h>
#include<limits.h>
#define V 5
int minKey(int k[],int m[]){
int min=INT_MAX,i,mi;
for(i=0;i<V;i++)if(!m[i]&&k[i]<min){min=k[i];mi=i;}
return mi;}
void printMST(int p[],int g[V][V]){
int i,c=0;
printf("Edge \tWeight\n");
for(i=1;i<V;i++){printf("%d - %d \t%d\n",p[i],i,g[i][p[i]]);c+=g[i][p[i]];}
printf("Total cost of MST: %d\n",c);}
void primMST(int g[V][V]){
int p[V],k[V],m[V]={0},i,c,u,v;

k[0]=0;p[0]=-1;
for(c=0;c<V-1;c++){
u=minKey(k,m);m[u]=1;
for(v=0;v<V;v++)
if(g[u][v]&&!m[v]&&g[u][v]<k[v]){p[v]=u;k[v]=g[u][v];}}

Lab Sheet 13: Knapsack Problem using Branch and Bound Technique

Aim:
To implement the 0/1 Knapsack Problem using the Branch and Bound technique in C to obtain the optimal solution.

Code:

#include<stdio.h>
struct I{int w,p;}it[10];
int mp=0;
float bound(int i,int w,int p,int n,int W){
float bp=p;int tw=w,j;
for(j=i;j<n&&tw+it[j].w<=W;j++){tw+=it[j].w;bp+=it[j].p;}
if(j<n)bp+=(W-tw)*(float)it[j].p/it[j].w;
return bp;}
void ks(int i,int w,int p,int n,int W){
if(w<=W&&p>mp)mp=p;
if(i==n)return;
if(bound(i+1,w,p,n,W)>mp)ks(i+1,w,p,n,W);
if(w+it[i].w<=W)ks(i+1,w+it[i].w,p+it[i].p,n,W);}
int main(){

printf("Enter knapsack capacity: ");

ks(0,0,0,n,W);

}

int j;

Lab Sheet 15: Case Study – Knapsack Problem using Multiple Techniques

Aim:
To study and compare different techniques used to solve the Knapsack Problem, including Greedy, Dynamic Programming, and Branch and Bound approaches.

Code:

#include<stdio.h>
int max(int a,int b){return a>b?a:b;}
void dp(int n,int W,int wt[],int v[]){
int K[20][20],i,w;
for(i=0;i<=n;i++)
for(w=0;w<=W;w++)
if(i==0||w==0)K[i][w]=0;
else if(wt[i-1]<=w)K[i][w]=max(v[i-1]+K[i-1][w-wt[i-1]],K[i-1][w]);
else K[i][w]=K[i-1][w];
printf("Maximum profit using Dynamic Programming: %d\n",K[n][W]);}
void gr(int n,int W,int wt[],int v[]){
float r[10],tp=0;int i,c=W;
for(i=0;i<n;i++)r[i]=(float)v[i]/wt[i];
for(i=0;i<n&&c>0;i++)
if(wt[i]<=c){c-=wt[i];tp+=v[i];}
else{tp+=r[i]*c;break;}
printf("Maximum profit using Greedy approach: %.2f\n",tp);}
int main(){
int n,W,i,wt[10],v[10];
printf("Enter number of items: ");

}
